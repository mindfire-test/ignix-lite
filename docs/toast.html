Toast

- Back to Docs

Usage


<ix-toast id="toast"></ix-toast>

<button onclick="toast.show({
  title: 'Saved',
  message: 'Your data has been updated.',
  intent: 'neutral'
})">
  Show Toast
</button>

Intent Variants


toast.show({
  title: 'Default',
  message: 'This is a neutral notification.',
  intent: 'neutral'
})

toast.show({
  title: 'Success',
  message: 'Your profile has been updated.',
  intent: 'success'
})

toast.show({
  title: 'Error',
  message: 'Unable to complete the action.',
  intent: 'danger'
})

toast.show({
  title: 'Warning',
  message: 'Check your inputs carefully.',
  intent: 'warning'
})

toast.show({
  title: 'Info',
  message: 'System update available.',
  intent: 'primary'
})

Motion Variants


toast.show({
  title: 'Saved',
  message: 'Your data has been updated.',
  variant: 'fade',
  intent: 'neutral'
})

toast.show({
  title: 'Saved',
  message: 'Your data has been updated.',
  variant: 'slide',
  intent: 'success'
})

toast.show({
  title: 'Error',
  message: 'Something failed.',
  variant: 'pop',
  intent: 'danger'
})

Action Toast


toast.show({
  title: 'Confirm',
  message: 'Do you want to continue?',
  intent: 'primary',
  variant: 'action'
})

Positions


<ix-toast id="topLeft" data-position="top-left"></ix-toast>
<ix-toast id="bottomRight" data-position="bottom-right"></ix-toast>
<ix-toast id="bottomLeft" data-position="bottom-left"></ix-toast>
<ix-toast id="topRight" data-position="top-right"></ix-toast>

topLeft.show({
  title: 'Top Left',
  message: 'Danger toast at top left.',
  intent: 'danger'
})

bottomRight.show({
  title: 'Bottom Right',
  message: 'Success toast at bottom right.',
  intent: 'success'
})

bottomLeft.show({
  title: 'Bottom Left',
  message: 'Warning toast at bottom left.',
  intent: 'warning'
})

topRight.show({
  title: 'Top Right',
  message: 'Primary toast at top right.',
  intent: 'primary'
})
